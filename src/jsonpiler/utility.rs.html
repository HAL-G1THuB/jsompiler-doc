9/// Decoding base64 variants.
10/// # Errors
11/// `Box<dyn Error(String)>` - If an invalid encoded value is passed, return `Err`
12#[expect(dead_code, reason = "todo")]
13pub(crate) fn de64(encoded: &str) -> Result<Vec<u8>, Box<dyn Error>> {
14  const ERR: &str = "Unreachable (de64)";
15  let mut decoded = Vec::new();
16  let mut buffer = 0u32;
17  let mut buffer_len = 0u32;
18  for ch in encoded.chars() {
19    if !('0'..='o').contains(&ch) {
20      return Err("Invalid character in input string.".into());
21    }
22    let val = u32::from(ch).checked_sub(48).ok_or(ERR)?;
23    buffer = (buffer << 6u32) | val;
24    buffer_len = buffer_len.checked_add(6).ok_or(ERR)?;
25    while buffer_len >= 8 {
26      let shift = buffer_len.checked_sub(8).ok_or(ERR)?;
27      let byte = u8::try_from(buffer >> shift)?;
28      decoded.push(byte);
29      buffer_len = shift;
30      buffer &= (1u32 << buffer_len).checked_sub(1).ok_or(ERR)?;
31    }
32  }
33  Ok(decoded)
34}
35/// Encoding base64 variants.
36/// # Errors
37/// Unreachable.
38#[expect(dead_code, reason = "todo")]
39pub(crate) fn en64(input: &[u8]) -> Result<String, &str> {
40  const ERR: &str = "Unreachable (en64)";
41  let mut encoded = String::new();
42  let chunks = input.chunks(3);
43  for chunk in chunks {
44    let b0 = chunk.first().unwrap_or(&0u8);
45    let b1 = chunk.get(1).unwrap_or(&0u8);
46    let b2 = chunk.get(2).unwrap_or(&0u8);
47    let enc1 = (b0 >> 2u8) & 0x3F;
48    let enc2 = ((b0 << 4u8) | (b1 >> 4u8)) & 0x3F;
49    let enc3 = ((b1 << 2u8) | (b2 >> 6u8)) & 0x3F;
50    let enc4 = b2 & 0x3F;
51    encoded.push(char::from_u32(u32::from(enc1).checked_add(48).ok_or(ERR)?).ok_or(ERR)?);
52    encoded.push(char::from_u32(u32::from(enc2).checked_add(48).ok_or(ERR)?).ok_or(ERR)?);
53    if chunk.len() >= 2 {
54      encoded.push(char::from_u32(u32::from(enc3).checked_add(48).ok_or(ERR)?).ok_or(ERR)?);
55    }
56    if chunk.len() == 3 {
57      encoded.push(char::from_u32(u32::from(enc4).checked_add(48).ok_or(ERR)?).ok_or(ERR)?);
58    }
59  }
60  Ok(encoded)
61}

68/// Escapes special characters in a string for proper JSON formatting.
69/// This method ensures that characters like quotes (`"`) and backslashes (`\`)
70/// are escaped in a way that conforms to the JSON specification.
71/// It also escapes control characters and non-ASCII characters using Unicode escapes.
72/// # Arguments
73/// * `s` - The string to be escaped.
74/// # Errors
75/// * `fmt::Error` - ...
76/// # Returns
77/// * `String` - The escaped string.
78pub(crate) fn escape_string(unescaped: &str) -> Result<String, fmt::Error> {
79  let mut escaped = String::new();
80  for ch in unescaped.chars() {
81    match ch {
82      '"' => write!(escaped, r#"\""#)?,
83      '\\' => write!(escaped, r"\\")?,
84      '\n' => write!(escaped, r"\n")?,
85      '\t' => write!(escaped, r"\t")?,
86      '\r' => write!(escaped, r"\r")?,
87      '\u{08}' => write!(escaped, r"\b")?,
88      '\u{0C}' => write!(escaped, r"\f")?,
89      u_ch if u_ch < '\u{20}' => write!(escaped, r"\u{:04x}", u32::from(u_ch))?,
90      _ => escaped.push(ch),
91    }
92  }
93  Ok(escaped)
94}
95/// Format error.
96/// # Errors
97/// `Box<dyn Error>` - Err is always returned.
98#[must_use]
99pub(crate) fn format_err(text: &str, pos: usize, ln: usize, source: &str) -> String {
100  const MSG1: &str = "\nError occurred on line: ";
101  const MSG2: &str = "\nError position:\n";
102  if source.is_empty() {
103    return format!("{text}{MSG1}{ln}{MSG2}Error: Empty input");
104  }
105  let len = source.len();
106  let idx = pos.min(len.saturating_sub(1));
107  let start = if idx == 0 {
108    0
109  } else {
110    match source[..idx].rfind('\n') {
111      None => 0,
112      Some(start_pos) => {
113        let Some(res) = start_pos.checked_add(1) else {
114          return format!("{text}{MSG1}{ln}{MSG2}Error: Overflow");
115        };
116        res
117      }
118    }
119  };
120  let end = match source[idx..].find('\n') {
121    None => len,
122    Some(end_pos) => {
123      let Some(res) = idx.checked_add(end_pos) else {
124        return format!("{text}{MSG1}{ln}{MSG2}Error: Overflow");
125      };
126      res
127    }
128  };
129  let ws = " ".repeat(idx.saturating_sub(start));
130  let result = &source[start..end];
131  format!("{text}{MSG1}{ln}{MSG2}{result}\n{ws}^")
132}
