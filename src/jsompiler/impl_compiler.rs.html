13  /// Assert condition.
14  #[expect(clippy::panic_in_result_fn, reason = "panic don't occurred")]
15  fn assert(&self, cond: bool, text: &str, obj: &Json) -> Result<(), Box<dyn Error>> {
16    if cond {
17      assert!(cond, "{text}");
18      Ok(())
19    } else {
20      Err(format_err(text, obj.pos, obj.line, &self.source).into())
21    }

23  /// Utility functions for binary operations
24  fn binary_op(
25    &mut self,
26    args: &[Json],
27    function: &mut String,
28    mnemonic: &str,
29    error_context: &str,
30  ) -> JFuncResult {
31    self.assert(
32      args.len() >= 3,
33      &format!("'{error_context}' requires at least two arguments"),
34      args.first().ok_or("Unreachable (binary_op)")?,
35    )?;
36    let result = self.eval_args(args.get(1..).ok_or("Unreachable (binary_op)")?, function)?;
37    match *result.first().ok_or("Unreachable (binary_op)")? {
38      JValue::Int(int) => writeln!(function, "  mov rax, {int}")?,
39      JValue::IntVar(ref iv) => writeln!(function, "  mov rax, qword ptr {iv}[rip]")?,
40      ref invalid @ (JValue::Array(_)
41      | JValue::ArrayVar(_)
42      | JValue::Bool(_)
43      | JValue::BoolVar(..)
44      | JValue::Float(_)
45      | JValue::FloatVar(_)
46      | JValue::FuncVar { .. }
47      | JValue::Null
48      | JValue::Object(_)
49      | JValue::ObjectVar(_)
50      | JValue::String(_)
51      | JValue::StringVar(_)) => {
52        return Err(
53          format!("'{error_context}' requires integer operands, but got '{invalid}'").into(),
54        );
55      }
56    }
57    for arg in result.get(1..).ok_or("Unreachable (binary_op)")? {
58      match *arg {
59        JValue::Int(int) => writeln!(function, "  {mnemonic} rax, {int}")?,
60        JValue::IntVar(ref var) => writeln!(function, "  {mnemonic} rax, qword ptr {var}[rip]")?,
61        JValue::Array(_)
62        | JValue::ArrayVar(_)
63        | JValue::Bool(_)
64        | JValue::BoolVar(..)
65        | JValue::Float(_)
66        | JValue::FloatVar(_)
67        | JValue::FuncVar { .. }
68        | JValue::Null
69        | JValue::Object(_)
70        | JValue::ObjectVar(_)
71        | JValue::String(_)
72        | JValue::StringVar(_) => {
73          return Err(
74            format!("'{error_context}' requires integer operands, but got '{arg}'").into(),
75          );
76        }
77      }
78    }
79    let ret = self.get_name()?;
80    writeln!(self.sect.bss, "  .lcomm {ret}, 8")?;
81    writeln!(function, "  mov qword ptr {ret}[rip], rax")?;
82    Ok(JValue::IntVar(ret))
83  }
84  /// Builds the assembly code from the parsed JSON.
85  ///
86  /// This function is the main entry point for the compilation process. It takes the parsed JSON,
87  /// sets up the initial function table, evaluates the JSON, and writes the resulting assembly
88  /// code to a file.
89  ///
90  /// # Arguments
91  ///
92  /// * `parsed` - The parsed JSON object.
93  /// * `json_file` - The name of the original JSON file.
94  /// * `filename` - The name of the file to write the assembly code to.
95  ///
96  /// # Returns
97  ///
98  /// * `Ok(Json)` - The result of the evaluation.
99  /// * `Err(Box<dyn Error>)` - If an error occurred during the compilation process.
100  ///
101  /// # Errors
102  ///
103  /// * `Box<dyn Error>` - If an error occurred during the compilation process.
104  #[inline]
105  pub fn build(
106    &mut self,
107    parsed: String,
108    json_file: &str,
109    filename: &str,
110  ) -> Result<(), Box<dyn Error>> {
111    let json = self.parse(parsed)?;
112    self.seed = 0;
113    self.register("=", Jsompiler::f_set_local);
114    self.register("$", Jsompiler::f_get_local);
115    self.register("+", Jsompiler::f_plus);
116    self.register("-", Jsompiler::f_minus);
117    self.register("message", Jsompiler::f_message);
118    self.register("begin", Jsompiler::f_begin);
119    let mut main_func = String::new();
120    self.sect = Section::default();
121    let result = self.eval(&json, &mut main_func)?;
122    writeln!(
123      main_func,
124      "  {}
125  call [qword ptr __imp_ExitProcess[rip]]
126  .seh_endproc",
127      match result.value {
128        JValue::Int(int) => format!("mov rcx, {int}"),
129        JValue::IntVar(ref var) => format!("mov rcx, qword ptr {var}[rip]"),
130        JValue::Array(_)
131        | JValue::ArrayVar(_)
132        | JValue::Bool(_)
133        | JValue::BoolVar(..)
134        | JValue::Float(_)
135        | JValue::FloatVar(_)
136        | JValue::FuncVar { .. }
137        | JValue::Null
138        | JValue::Object(_)
139        | JValue::ObjectVar(_)
140        | JValue::String(_)
141        | JValue::StringVar(_) => String::from("xor ecx, ecx"),
142      }
143    )?;
144    self.write_file(&main_func, filename, json_file)?;
145    Ok(())
146  }

155  /// Evaluates a JSON object.
156  fn eval(&mut self, parsed: &Json, function: &mut String) -> JResult {
157    let JValue::Array(ref list) = parsed.value else {
158      return Ok(parsed.clone());
159    };
160    self.assert(!list.is_empty(), "An function call cannot be an empty list.", parsed)?;
161    match list.first().ok_or("Unreachable (eval)")?.value {
162      JValue::String(ref cmd) => {
163        if cmd == "lambda" {
164          let mut func_buffer = String::new();
165          let result = Ok(self.eval_lambda(parsed, &mut func_buffer)?);
166          self.sect.text.push_str(&func_buffer);
167          result
168        } else if let Some(func) = self.f_table.get(cmd.as_str()) {
169          Ok(obj_json(func(self, list, function)?, list.first().ok_or("Unreachable (eval)")?))
170        } else {
171          self.err_compile(
172            &format!("Function {cmd} is undefined."),
173            list.first().ok_or("Unreachable (eval)")?,
174          )
175        }
176      }
177      JValue::Array(_) => {
178        let mut func_buffer = String::new();
179        let tmp = self.vars.clone();
180        let lambda =
181          self.eval_lambda(list.first().ok_or("Unreachable (eval)")?, &mut func_buffer)?;
182        let JValue::FuncVar { name: ref n, ret: ref re, .. } = lambda.value else {
183          return self.err_compile("InternalError: 'lambda' don't return lambda object.", &lambda);
184        };
185        self.sect.text.push_str(&func_buffer);
186        writeln!(function, "  call {n}")?;
187        self.vars = tmp;
188        match **re {
189          JValue::IntVar(_) => {
190            let na = self.get_name()?;
191            writeln!(self.sect.bss, "  .lcomm {na}, 8")?;
192            writeln!(function, "  mov qword ptr {na}[rip], rax")?;
193            Ok(obj_json(JValue::IntVar(na), &lambda))
194          }
195          JValue::Int(_) => {
196            let na = self.get_name()?;
197            writeln!(self.sect.bss, "  .lcomm {na}, 8")?;
198            writeln!(function, "  mov qword ptr {na}[rip], rax")?;
199            Ok(obj_json(JValue::IntVar(na), &lambda))
200          }
201          JValue::Array(_)
202          | JValue::ArrayVar(_)
203          | JValue::Bool(_)
204          | JValue::BoolVar(..)
205          | JValue::Float(_)
206          | JValue::FloatVar(_)
207          | JValue::FuncVar { .. }
208          | JValue::Null
209          | JValue::Object(_)
210          | JValue::ObjectVar(_)
211          | JValue::String(_)
212          | JValue::StringVar(_) => Ok(Json::default()),
213        }
214      }
215      JValue::ArrayVar(_)
216      | JValue::Bool(_)
217      | JValue::BoolVar(..)
218      | JValue::Float(_)
219      | JValue::FloatVar(_)
220      | JValue::FuncVar { .. }
221      | JValue::Int(_)
222      | JValue::IntVar(_)
223      | JValue::Null
224      | JValue::Object(_)
225      | JValue::ObjectVar(_)
226      | JValue::StringVar(_) => self.err_compile(
227        "The first element of an evaluation list requires a function name or a lambda object.",
228        parsed,
229      ),
230    }
231  }
232  /// Evaluate arguments.
233  fn eval_args(
234    &mut self,
235    args: &[Json],
236    function: &mut String,
237  ) -> Result<Vec<JValue>, Box<dyn Error>> {
238    let mut result = vec![];
239    for arg in args {
240      result.push(self.eval(arg, function)?.value);
241    }
242    Ok(result)
243  }
244  /// Evaluates a lambda function definition.
245  fn eval_lambda(&mut self, func: &Json, function: &mut String) -> JResult {
246    self.assert(matches!(func.value, JValue::Array(_)), "Invalid function definition", func)?;
247    let JValue::Array(ref func_list) = func.value else {
248      return self.err_compile("Invalid function definition", func);
249    };
250    self.assert(func_list.len() >= 3, "Invalid function definition", func)?;
251    let lambda = func_list.first().ok_or("Unreachable (eval_lambda)")?;
252    self.assert(
253      matches!(lambda.value, JValue::String(ref st) if st == "lambda"),
254      "The first element of a lambda list requires \"lambda\".",
255      lambda,
256    )?;
257    let JValue::Array(ref params) = func_list.get(1).ok_or("Unreachable (eval_lambda)")?.value
258    else {
259      return self.err_compile(
260        "The second element of a lambda list requires an argument list.",
261        func_list.get(1).ok_or("Unreachable (eval_lambda)")?,
262      );
263    };
264    self.assert(
265      params.is_empty(),
266      "PARAMS IS TODO.",
267      func_list.get(1).ok_or("Unreachable (eval_lambda)")?,
268    )?;
269    let n = self.get_name()?;
270    writeln!(
271      function,
272      ".section .text${},\"x\"
273.seh_proc {n}
274{n}:
275  push rbp
276  .seh_pushreg rbp
277  mov rbp, rsp
278  .seh_setframe rbp, 0
279  sub rsp, 32
280  .seh_stackalloc 32
281  .seh_endprologue
282  .seh_handler .L_SEH_HANDLER, @except",
283      &n[3..]
284    )?;
285    let result = self
286      .eval_args(func_list.get(2..).ok_or("Unreachable (eval_lambda)")?, function)?
287      .last()
288      .ok_or_else(|| format_err("Empty lambda body", lambda.pos, lambda.line, &self.source))?
289      .clone();
290    writeln!(
291      function,
292      "  {}
293  add rsp, 32
294  leave
295  ret
296  .seh_endproc",
297      match result {
298        JValue::Int(int) => format!("mov rax, {int}"),
299        JValue::IntVar(ref var) => format!("mov rax, qword ptr {var}[rip]"),
300        JValue::Array(_)
301        | JValue::ArrayVar(_)
302        | JValue::Bool(_)
303        | JValue::BoolVar(..)
304        | JValue::Float(_)
305        | JValue::FloatVar(_)
306        | JValue::FuncVar { .. }
307        | JValue::Null
308        | JValue::Object(_)
309        | JValue::ObjectVar(_)
310        | JValue::String(_)
311        | JValue::StringVar(_) => "xor eax, eax".into(),
312      }
313    )?;
314    Ok(obj_json(JValue::FuncVar { name: n, params: params.clone(), ret: Box::new(result) }, lambda))
315  }
316  /// Evaluates a 'begin' block.
317  fn f_begin(&mut self, args: &[Json], function: &mut String) -> JFuncResult {
318    self.assert(
319      args.len() >= 2,
320      "'begin' requires at least one arguments",
321      args.first().ok_or("Unreachable (begin)")?,
322    )?;
323    let begin = args.first().ok_or("Unreachable (f_begin)")?;
324    Ok(
325      self
326        .eval_args(args.get(1..).ok_or("Unreachable (f_begin)")?, function)?
327        .last()
328        .ok_or_else(|| format_err("Empty lambda body", begin.pos, begin.line, &self.source))?
329        .clone(),
330    )
331  }
332  /// Gets the value of a local variable.
333  fn f_get_local(&mut self, args: &[Json], _: &mut String) -> JFuncResult {
334    self.assert(
335      args.len() == 2,
336      "'$' requires one argument",
337      args.first().ok_or("Unreachable (f_get_local)")?,
338    )?;
339    let var = args.get(1).ok_or("Unreachable (f_get_local)")?;
340    let JValue::String(ref var_name) = var.value else {
341      return self.err_func("Variable name requires compile-time fixed string", var);
342    };
343    if let Some(value) = self.vars.get(var_name) {
344      Ok(value.clone())
345    } else {
346      self.err_func(&format!("Undefined variables: '{var_name}'"), var)
347    }
348  }
349  /// Displays a message box.
350  fn f_message(&mut self, args: &[Json], function: &mut String) -> JFuncResult {
351    self.assert(
352      args.len() == 3,
353      "'message' requires two arguments",
354      args.first().ok_or("Unreachable (f_message)")?,
355    )?;
356    let title_json = args.get(1).ok_or("Unreachable (f_message)")?;
357    let title = match self.eval(title_json, function)?.value {
358      JValue::String(st) => {
359        let name = self.get_name()?;
360        writeln!(self.sect.data, "  {name}: .string \"{st}\"")?;
361        name
362      }
363      JValue::StringVar(var) => var,
364      JValue::Array(_)
365      | JValue::ArrayVar(_)
366      | JValue::Bool(_)
367      | JValue::BoolVar(..)
368      | JValue::Float(_)
369      | JValue::FloatVar(_)
370      | JValue::FuncVar { .. }
371      | JValue::Int(_)
372      | JValue::IntVar(_)
373      | JValue::Null
374      | JValue::Object(_)
375      | JValue::ObjectVar(_) => {
376        return self.err_func("The first argument of message must be a string", title_json);
377      }
378    };
379    let msg_json = args.get(2).ok_or("Unreachable (f_message)")?;
380    let msg = match self.eval(msg_json, function)?.value {
381      JValue::String(st) => {
382        let name = self.get_name()?;
383        writeln!(self.sect.data, "  {name}: .string \"{st}\"")?;
384        name
385      }
386      JValue::StringVar(var) => var,
387      JValue::Array(_)
388      | JValue::ArrayVar(_)
389      | JValue::Bool(_)
390      | JValue::BoolVar(..)
391      | JValue::Float(_)
392      | JValue::FloatVar(_)
393      | JValue::FuncVar { .. }
394      | JValue::Int(_)
395      | JValue::IntVar(_)
396      | JValue::Null
397      | JValue::Object(_)
398      | JValue::ObjectVar(_) => {
399        return self.err_func("The second argument of message must be a string", msg_json);
400      }
401    };
402    let wtitle = self.get_name()?;
403    let wmsg = self.get_name()?;
404    let ret = self.get_name()?;
405    writeln!(
406      self.sect.bss,
407      "  .lcomm {wtitle}, 8
408  .lcomm {wmsg}, 8
409  .lcomm {ret}, 8"
410    )?;
411    write!(
412      function,
413      include_str!("message.s"),
414      msg, msg, wmsg, title, title, wtitle, wmsg, wtitle, ret, wmsg, wtitle
415    )?;
416    Ok(JValue::IntVar(ret))
417  }

426  /// Sets a local variable.
427  fn f_set_local(&mut self, args: &[Json], function: &mut String) -> JFuncResult {
428    self.assert(
429      args.len() == 3,
430      "'=' requires two arguments",
431      args.first().ok_or("Unreachable (f_set_local)")?,
432    )?;
433    let JValue::String(ref var_name) = args.get(1).ok_or("Unreachable (f_set_local)")?.value else {
434      return self.err_func(
435        "Variable name requires compile-time fixed strings",
436        args.get(1).ok_or("Unreachable (f_set_local)")?,
437      );
438    };
439    let result = self.eval(args.get(2).ok_or("Unreachable (f_set_local)")?, function)?;
440    let n = self.get_name()?;
441    match result.value {
442      JValue::String(ref st) => {
443        writeln!(self.sect.data, "  {n}: .string \"{st}\"")?;
444        self.vars.insert(var_name.clone(), JValue::StringVar(n.clone()));
445        Ok(JValue::StringVar(n))
446      }
447      JValue::StringVar(ref sv) => {
448        self.vars.insert(var_name.clone(), JValue::StringVar(sv.clone()));
449        Ok(result.value)
450      }
451      JValue::Array(_)
452      | JValue::ArrayVar(_)
453      | JValue::Bool(_)
454      | JValue::BoolVar(..)
455      | JValue::Float(_)
456      | JValue::FloatVar(_)
457      | JValue::FuncVar { .. }
458      | JValue::Int(_)
459      | JValue::IntVar(_)
460      | JValue::Null
461      | JValue::Object(_)
462      | JValue::ObjectVar(_) => Err("Assignment to an unimplemented type".into()),
463    }
464  }
465  /// Generates a unique name for internal use.
466  fn get_name(&mut self) -> Result<String, Box<dyn Error>> {
467    let Some(res) = self.seed.checked_add(1) else { return Err("Seed Overflow".into()) };
468    self.seed = res;
469    Ok(format!(".LC{:x}", self.seed))
470  }
471  /// Registers a function in the function table.
472  fn register(&mut self, name: &str, func: JFunc<Self>) {
473    self.f_table.insert(name.into(), func);
474  }
475  /// Writes the compiled assembly code to a file.
476  fn write_file(&self, main_func: &str, filename: &str, json_file: &str) -> io::Result<()> {
477    let file = File::create(filename)?;
478    let mut writer = BufWriter::new(file);
479    writer.write_all(format!(".file \"{json_file}\"\n.intel_syntax noprefix\n").as_bytes())?;
480    writer.write_all(include_bytes!("data.s"))?;
481    writer.write_all(self.sect.data.as_bytes())?;
482    writer.write_all(include_bytes!("bss.s"))?;
483    writer.write_all(self.sect.bss.as_bytes())?;
484    writer.write_all(include_bytes!("start.s"))?;
485    writer.write_all(main_func.as_bytes())?;
486    writer.write_all(include_bytes!("text.s"))?;
487    writer.write_all(self.sect.text.as_bytes())?;
488    writer.flush()?;
489    Ok(())
490  }
